要素の取得

Pure JavaScript

      // 構文
      document.querySelector('[css セレクタ]'); // 最初の一つを取得
      document.querySelectorAll('[css セレクタ]'); // すべて取得

      // 例
      document.querySelector('#main .container'); // 最初の一つを取得
      document.querySelectorAll('#main .container'); // すべて取得

jQuery

      // 構文
      $('[CSS のセレクタ]')

      // 例
      $('#main .container h1')

イベントの設定

Pure JavaScript

      // 構文
      [要素].addEventListener('[イベント名]', function () {
        // イベントが発火したときの処理
      });

      // 例
      var button document.getElementById('button');
      button.addEventListener('click', function () {
        alert('クリックされた！');
      });

jQuery

      // 構文
      [jQuery 要素].[イベント名](function () {
        // イベントが発火したときの処理
      });

      // 例
      $('#button').click(function () {
        alert('クリックされた！');
      });

代表的なイベントの種類

イベント名 発火タイミング
ready DOMツリーの構築が完了したときに発火
load リソースの読み込みが完了した時に発火
error リソースの読み込みに失敗した時に発火
scroll ドキュメントがスクロールした時に発火
resize ウィンドウサイズが変更された時に発火
click 要素がクリックされた時に発火
dblclick 要素がダブルクリックされた時に発火
mousedown 要素上でマウスが押された時に発火
mouseup 要素上でマウスが押され、上がった時に発火
mousemove 要素上でマウスが移動している時に発火
mouseover マウスが要素に入った時に発火。子要素でも発火
mouseout マウスが要素から外れた時に発火。子要素でも発火
focus 要素がフォーカスを得た時に発火
blur 要素がフォーカスを失った時に発火
change 要素の内容が変更され、フォーカスが外れた時に発火
input 要素の内容が変更された時に発火
submit フォームが送信された時に発火
keydown キーを押しこんだ時に発火。押している間ずっと発火
keyup キーが上がった時に発火

読み込み完了

      // DOM がすべて読み込まれたときに発火
      $(document).ready(function() {
        // 処理を書く
      });

      // ready の省略型
      $(function(){
        // 処理を書く
      });

      // 画像など全てのリソースが読み込まれた発火
      $(window).load(function(){
        // 処理を書く
      });

属性の操作

属性の取得

      // 構文
      $('[セレクタ]').attr('[属性名]');

      // 例
      $('#textbox').attr('value');

属性の設定

      // 構文
      $('[セレクタ]').attr('[属性名]', '[属性値]');

      // 例
      $('#textbox').attr('value', 'jsから入力！');

属性の削除

      // 構文
      $('[セレクタ]').removeAttr('[属性名]');

      // 例
      $('#button').removeAttr('disabled');

スタイル関連の操作

CSS の追加

      // 構文
      $('[セレクタ]').css('[プロパティ名]', '[値]');

      // 例
      $('.button').css('color', 'red');

CSS の削除

      // 構文
      $('[セレクタ]').css('[プロパティ名]', '');

      // 例
      $('.button').css('color', '');

クラスの追加

      // 構文
      $('[セレクタ]').addClass('[クラス名]');

      // 例
      $('#button').addClass('active');

クラスの削除

      // 構文
      $('[セレクタ]').removeClass('[クラス名]');

      // 例
      $('#button').removeClass('active');

要素の操作

要素の追加

      // 構文
      $('[セレクタ]').append('[要素]'); // 要素内の最後に追加
      $('[セレクタ]').prepend('[要素]'); // 要素内の最初に追加

      $('[セレクタ]').after('[要素]'); // 要素の次に追加
      $('[セレクタ]').before('[要素]'); // 要素の前に追加

      // 例
      $('#list').append('
  • 追加されたよ！
    • ');
      $('#list').prepend('
  • 追加されたよ！
    • ');

      $('#list').after('
    追加されたよ！
    ');
      $('#list').before('
    追加されたよ！
    ');

    要素の削除

          // 構文
      $('[セレクタ]').remove(); // 要素を削除
      $('[セレクタ]').empty(); // 要素の中身を削除

      // 例
      $('.container').remove();
      $('.container').empty();
